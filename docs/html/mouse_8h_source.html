61// covers what action mode the mouse is in after a mouse down event: either dragging one or more icons, or selecting via lasso, or neither
62typedef enum MouseMode
63{
64 mouseFree = 0,
65 mouseSelect = 1, // user has clicked on icon(s), but not moved mouse enough to start drag
66 mouseDoubleclick = 2,
67 mouseDrag = 3, // user clicked on icon(s) and moved mouse enough to start drag mode
68 mouseLasso = 4, // nothing under cursor, button down, ready to start drawing a lasso
69 mouseLassoInProgress = 5, // user has moved mouse from origin point with mouse down, lasso is actively being drawn
70 mouseDownOnControl = 6, // mouse was clicked within bounds of a control. Mouse button is not released.
71 mouseDragTitle = 7,
72 mouseMenuOpen = 8, // right click opened a menu, and menu has not been closed yet.
73 mouseResizeUp = 9,
74 mouseResizeRight = 10,
75 mouseResizeDown = 11,
76 mouseResizeLeft = 12,
77 mouseResizeDownRight = 13,
78} MouseMode;
