108struct Window
109{
110 uint8_t id_; // reserved. Not currently used.
111 int8_t display_order_; // 0 = active window, ascending order from there. maintained by system.
112 unsigned long user_data_; // 32 bits for use of programs. The system will not process this field.
113 char* title_;
114 window_type type_;
115 window_state state_; // read-only: whether the window is currently hidden, minimized, normal/window sized, or maximized
116 Bitmap* bitmap_; // on-screen bitmap covering the visible portion of window
117 Bitmap* buffer_bitmap_; // off-screen bitmap covering the visible portion of window
118// struct Region* global_region_;
119// struct Region* content_region_; // portion of window that will contain content. excludes the borders and title bar
120// struct Region* iconbar_region_; // region for the iconbar, if any
121// struct Region* titlebar_region_;
122 Rectangle overall_rect_; // the rect describing the total area of the window
123 Rectangle content_rect_; // the rect describing the content area of the window
124 Rectangle titlebar_rect_; // the rect describing the titlebar area
125 Rectangle grow_left_rect_; // the rect defining the area in which a click/drag will resize window
126 Rectangle grow_right_rect_; // the rect defining the area in which a click/drag will resize window
127 Rectangle grow_top_rect_; // the rect defining the area in which a click/drag will resize window
128 Rectangle grow_bottom_rect_; // the rect defining the area in which a click/drag will resize window
129 boolean is_backdrop_; // true if this is the backdrop (desktop) window
130 boolean visible_; // is the window visible?
131 boolean active_; // keep 1 window as the active one. only active windows get regular updates
132 boolean changes_to_save_; // starts at false; if window is resized or repositioned, is set to true. used to know if we have to save out to icon file when window is closed.
133 boolean can_resize_; // if true, window can be stretched or shrunk. If false, the width_ and height_ will be locked.
134 signed int x_; // horizontal coordinate when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
135 signed int y_; // vertical coordinate when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
136 signed int width_; // width of window when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
137 signed int height_; // height of window when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
138 signed int min_width_; // minimum width of window when in window-sized (normal) mode.
139 signed int min_height_; // minimum height of window when in window-sized (normal) mode.
140 signed int max_width_; // maximum width of window when in window-sized (normal) mode. If > 0, the window will not maximize. Default 0.
141 signed int max_height_; // maximum height of window when in window-sized (normal) mode. If > 0, the window will not maximize. Default 0.
142 signed int inner_width_; // space available inside the window, accounting for border thicknesses
143 signed int inner_height_; // space available inside the window, accounting for border thicknesses and title bar
144 signed int content_left_; // of the raw x pos of the window (non-gzz), the x pos where window should start rendering content. =gzz_left_ until window is scrolled leftwards
145 signed int content_top_; // of the raw y pos of the window (non-gzz), the y pos where window should start rendering content. =gzz_top_ until window is scrolled down
146 signed int required_inner_width_; // greater of current inner_width or H space required inside the window to display all content. If greater than H space, a scroller is needed.
147 signed int required_inner_height_; // greater of current inner_height or V space required inside the window to display all content. If greater than V space, a scroller is needed.
148 boolean h_scroller_visible_;
149 boolean v_scroller_visible_;
150 Bitmap* pattern_; // optional pattern used for filling the window content rect background on refresh.
151 Window* parent_window_; // can be NULL. used for requesters that are spawned from a specific window.
152 Window* child_window_; // can be NULL. used when a window spawns a requester. (This is the requester). NULLs out again when requester is closed.
153 Control* root_control_; // first control in the window
154
155// MouseTracker* mouse_tracker_; // tracks mouse up/down points within this window
156// Window* zoom_to_window_; // the window that contains the zoom_to_file, so we can get offset to global screen coords
157// unsigned short zoom_x_[WIN_ZOOM_RECT_COUNT]; // Used to plot the coords for zoom rects when opening/closing window
158// unsigned short zoom_y_[WIN_ZOOM_RECT_COUNT]; // Used to plot the coords for zoom rects when opening/closing window
159// unsigned short zoom_w_[WIN_ZOOM_RECT_COUNT]; // Used to plot the coords for zoom rects when opening/closing window
160// unsigned short zoom_h_[WIN_ZOOM_RECT_COUNT]; // Used to plot the coords for zoom rects when opening/closing window
161// struct Menu* menu_;
162// unsigned char* keyboard_buffer_; // used by key detection
163// unsigned short keyboard_buf_pos_; // used by key detection
164};
