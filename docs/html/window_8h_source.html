14/* about this library: Window
15 *
16 * This provides functionality for working with windows
17 *
18 * See also: control.h
19 *
20 *** things this library needs to be able to do
21 * Create a new window instance based on a Theme object and other parameters
22 * Render a window on the screen
23 * Receive and manage events
24 * Identify which control is affected by a mouse click or mouse move event
25 * Manage state of all controls on the window
26 * Handle resizing/minimizing/maximizing of window
27 * Reposition controls on window in response to resize events
28 *
29 *
30 * STRETCH GOALS

153 uint8_t pen_color_; // Color index of the "pen", for drawing functions
154 Font* pen_font_; // Font to be used by the "pen", for drawing functions
155 Rectangle overall_rect_; // the local rect describing the total area of the window

163 Rectangle grow_bottom_rect_; // the local rect defining the area in which a click/drag will resize window
164 Rectangle grow_bottom_right_rect_; // the local rect defining the area in which a click/drag will resize window
165 bool show_iconbar_; // true if the iconbar area should be rendered
166 bool is_backdrop_; // true if this is the backdrop (desktop) window
167 bool visible_; // is the window visible?
168 bool active_; // keep 1 window as the active one. only active windows get regular updates
169 bool changes_to_save_; // starts at false; if window is resized or repositioned, is set to true. used to know if we have to save out to icon file when window is closed.
170 bool can_resize_; // if true, window can be stretched or shrunk. If false, the width_ and height_ will be locked.
171 bool invalidated_; // if true, the window needs to be completely re-rendered on the next render pass
172 bool titlebar_invalidated_; // if true, the titlebar needs to be redrawn and reblitted on the next render pass
173 int16_t x_; // current global horizontal coordinate
174 int16_t y_; // current global vertical coordinate
175 int16_t proposed_x_; // during a window drag or resize event, the potential future global horizontal coordinate
176 int16_t proposed_y_; // during a window drag or resize event, the potential future global vertical coordinate
177 Rectangle global_rect_; // the global rect describing the total area of the window
178 int16_t width_; // current width of window
179 int16_t height_; // current height of window
180 int16_t norm_x_; // global horizontal coordinate when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
181 int16_t norm_y_; // global vertical coordinate when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
182 int16_t norm_width_; // width of window when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
183 int16_t norm_height_; // height of window when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
184 int16_t min_width_; // minimum width of window when in window-sized (normal) mode.
185 int16_t min_height_; // minimum height of window when in window-sized (normal) mode.
186 int16_t max_width_; // maximum width of window when in window-sized (normal) mode. If > 0, the window will not maximize. Default 0.
187 int16_t max_height_; // maximum height of window when in window-sized (normal) mode. If > 0, the window will not maximize. Default 0.
188 int16_t inner_width_; // space available inside the content area, accounting for border thicknesses
189 int16_t inner_height_; // space available inside the content area, accounting for border thicknesses and title bar
190 int16_t avail_title_width_; // available pixel width for the title to be rendered, based on delta between title x offset and left-most titlebar control
191 int16_t content_left_; // of the raw x pos of the window (non-gzz), the x pos where window should start rendering content. =gzz_left_ until window is scrolled leftwards
192 int16_t content_top_; // of the raw y pos of the window (non-gzz), the y pos where window should start rendering content. =gzz_top_ until window is scrolled down
193 int16_t required_inner_width_; // greater of current inner_width or H space required inside the window to display all content. If greater than H space, a scroller is needed.
194 int16_t required_inner_height_; // greater of current inner_height or V space required inside the window to display all content. If greater than V space, a scroller is needed.
195 bool h_scroller_visible_;
196 bool v_scroller_visible_;
197 Bitmap* pattern_; // optional pattern used for filling the window content rect background on refresh.
198 Window* parent_window_; // can be NULL. used for requesters that are spawned from a specific window.
199 Window* child_window_; // can be NULL. used when a window spawns a requester. (This is the requester). NULLs out again when requester is closed.
200 Control* root_control_; // first control in the window
201 Control* selected_control_; // the currently selected control for the window. Only 1 can be selected per window. No guarantee that any are selected.
202 Rectangle clip_rect_[WIN_MAX_CLIP_RECTS]; // one or more clipping rects; determines which parts of window need to be blitted to the main screen
203 int16_t clip_count_; // number of clip rects the window is currently tracking
204 Rectangle damage_rect_[4]; // 0 to 4 rects that describe to other windows under this one, which parts of the screen were previously covered by this window (prior to a move or resize)
205 int16_t damage_count_; // number of damage rects the window is currently tracking
206 void (*event_handler_)(EventRecord*); // function that will be called by the system when an event related to the window is encountered.
207 Menu* menu_[WIN_MENU_MAX_GROUPS]; // non-permanent containers for menu structures; will be used for first, 2nd, 3rd, and 4th level menus as used in the window.
208 int16_t current_menu_level_; // index to menu_[]; starts out at menu_no_menu; when a menu is opened, it goes to menu_level_0; increases with each submenu. Resets to menu_no_men uon close of menu.
209// Window* zoom_to_window_; // the window that contains the zoom_to_file, so we can get offset to global screen coords
210// int16_t zoom_x_[WIN_ZOOM_RECT_COUNT]; // Used to plot the coords for zoom rects when opening/closing window
211// int16_t zoom_y_[WIN_ZOOM_RECT_COUNT]; // Used to plot the coords for zoom rects when opening/closing window

214// unsigned char* keyboard_buffer_; // used by key detection
215// uint16_t keyboard_buf_pos_; // used by key detection
216};
