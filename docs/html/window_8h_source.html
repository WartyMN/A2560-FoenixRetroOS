14/* about this library: Window
15 *
16 * This provides functionality for working with windows
17 *
18 * See also: control.h
19 *
20 *** things this library needs to be able to do
21 * Create a new window instance based on a Theme object and other parameters
22 * Render a window on the screen
23 * Receive and manage events
24 * Identify which control is affected by a mouse click or mouse move event
25 * Manage state of all controls on the window

115struct Window
116{
117 uint8_t id_; // reserved. Not currently used.
118 int8_t display_order_; // 0 = active window, ascending order from there. maintained by system.
119 uint32_t user_data_; // 32 bits for use of programs. The system will not process this field.
120 char* title_;
121 window_type type_;
122 window_state state_; // read-only: whether the window is currently hidden, minimized, normal/window sized, or maximized
123 Bitmap* bitmap_; // on-screen bitmap covering the visible portion of window
124 Bitmap* buffer_bitmap_; // off-screen bitmap covering the visible portion of window
125// struct Region* global_region_;
126// struct Region* content_region_; // portion of window that will contain content. excludes the borders and title bar
127// struct Region* iconbar_region_; // region for the iconbar, if any
128// struct Region* titlebar_region_;
129 int16_t pen_x_; // H position relative to the content_rect_, of the "pen", for drawing functions
130 int16_t pen_y_; // V position relative to the content_rect_, of the "pen", for drawing functions
131 uint8_t pen_color_; // Color index of the "pen", for drawing functions
132 Font* pen_font_; // Font to be used by the "pen", for drawing functions
133 Rectangle overall_rect_; // the local rect describing the total area of the window
134 Rectangle titlebar_rect_; // the local rect describing the titlebar area
135 Rectangle iconbar_rect_; // the local rect describing the optional iconbar area
136 Rectangle content_rect_; // the local rect describing the content area of the window
137 Rectangle grow_left_rect_; // the local rect defining the area in which a click/drag will resize window
138 Rectangle grow_right_rect_; // the local rect defining the area in which a click/drag will resize window
139 Rectangle grow_top_rect_; // the local rect defining the area in which a click/drag will resize window
140 Rectangle grow_bottom_rect_; // the local rect defining the area in which a click/drag will resize window
141 bool show_iconbar_; // true if the iconbar area should be rendered
142 bool is_backdrop_; // true if this is the backdrop (desktop) window
143 bool visible_; // is the window visible?
144 bool active_; // keep 1 window as the active one. only active windows get regular updates
145 bool changes_to_save_; // starts at false; if window is resized or repositioned, is set to true. used to know if we have to save out to icon file when window is closed.
146 bool can_resize_; // if true, window can be stretched or shrunk. If false, the width_ and height_ will be locked.
147 bool invalidated_; // if true, the window needs to be completely re-rendered on the next render pass
148 bool titlebar_invalidated_; // if true, the titlebar needs to be redrawn and reblitted on the next render pass
149 int16_t x_; // global horizontal coordinate when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
150 int16_t y_; // global vertical coordinate when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
151 Rectangle global_rect_; // the global rect describing the total area of the window
152 int16_t width_; // width of window when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
153 int16_t height_; // height of window when in window-sized (normal) mode. Not adjusted when window is minimized or maximized.
154 int16_t min_width_; // minimum width of window when in window-sized (normal) mode.
155 int16_t min_height_; // minimum height of window when in window-sized (normal) mode.
156 int16_t max_width_; // maximum width of window when in window-sized (normal) mode. If > 0, the window will not maximize. Default 0.
157 int16_t max_height_; // maximum height of window when in window-sized (normal) mode. If > 0, the window will not maximize. Default 0.
158 int16_t inner_width_; // space available inside the content area, accounting for border thicknesses
159 int16_t inner_height_; // space available inside the content area, accounting for border thicknesses and title bar
160 int16_t avail_title_width_; // available pixel width for the title to be rendered, based on delta between title x offset and left-most titlebar control
161 int16_t content_left_; // of the raw x pos of the window (non-gzz), the x pos where window should start rendering content. =gzz_left_ until window is scrolled leftwards
162 int16_t content_top_; // of the raw y pos of the window (non-gzz), the y pos where window should start rendering content. =gzz_top_ until window is scrolled down
163 int16_t required_inner_width_; // greater of current inner_width or H space required inside the window to display all content. If greater than H space, a scroller is needed.
164 int16_t required_inner_height_; // greater of current inner_height or V space required inside the window to display all content. If greater than V space, a scroller is needed.
165 bool h_scroller_visible_;
166 bool v_scroller_visible_;
167 Bitmap* pattern_; // optional pattern used for filling the window content rect background on refresh.
168 Window* parent_window_; // can be NULL. used for requesters that are spawned from a specific window.
169 Window* child_window_; // can be NULL. used when a window spawns a requester. (This is the requester). NULLs out again when requester is closed.
170 Control* root_control_; // first control in the window
171 Control* selected_control_; // the currently selected control for the window. Only 1 can be selected per window. No guarantee that any are selected.
172 ClipRect* clip_rect_; // one or more clipping rects; determines which parts of window need to be blitted to the main screen
173
174// MouseTracker* mouse_tracker_; // tracks mouse up/down points within this window
175// Window* zoom_to_window_; // the window that contains the zoom_to_file, so we can get offset to global screen coords
176// int16_t zoom_x_[WIN_ZOOM_RECT_COUNT]; // Used to plot the coords for zoom rects when opening/closing window
177// int16_t zoom_y_[WIN_ZOOM_RECT_COUNT]; // Used to plot the coords for zoom rects when opening/closing window
178// int16_t zoom_w_[WIN_ZOOM_RECT_COUNT]; // Used to plot the coords for zoom rects when opening/closing window
179// int16_t zoom_h_[WIN_ZOOM_RECT_COUNT]; // Used to plot the coords for zoom rects when opening/closing window
180// struct Menu* menu_;
181// unsigned char* keyboard_buffer_; // used by key detection
182// uint16_t keyboard_buf_pos_; // used by key detection
183};
